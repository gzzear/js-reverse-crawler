# Excel字段与数据包映射关系

## 完整映射表

| Excel列 | 字段名 | 数据包路径 | 说明 | 示例值 |
|---------|--------|-----------|------|--------|
| A | 产品ID | `goods.goodsID` 或 `goods.goods_id` | 商品ID | 872145097620 |
| B | 标题 | `goods.goodsName` 或 `goods.goods_name` 或 `goods.shareDesc` | 商品名称 | 养殖兽用奶瓶小猪奶瓶... |
| C | 商品描述 | `goods.goodsProperty[]` | 从商品属性构建 | 品牌：其他\n主材质：其他 |
| D | 分类ID | `goods.catID` 或 `goods.cat_id` | 商品分类ID | 24941 |
| E | 价格 | `skus[].groupPrice` 或 `skus[].group_price` | 所有SKU价格，用\|\|分隔 | 21.57\|\|15\|\|29.62 |
| F | 规格一 | `skus[].specs[0].spec_key` | 第一个规格维度名称 | 属性 |
| G | 规格一内容 | `skus[].specs[0].spec_value` | 第一个规格所有值，用\|\|分隔 | 250ml奶瓶1个+2个备用奶嘴\|\|... |
| H | 规格二 | `skus[].specs[1].spec_key` | 第二个规格维度名称（如果有） | 尺码（或空） |
| I | 规格二内容 | `skus[].specs[1].spec_value` | 第二个规格所有值，用\|\|分隔 | S\|\|M\|\|L（或空） |
| J | 规格图片 | `skus[].thumbUrl` 或 `skus[].thumb_url` | SKU图片（嵌入显示） | 🖼️ 图片显示 |
| K | 图片 | `goods.viewImageData[]` 或 `goods.gallery[].url` | 主图（嵌入显示） | 🖼️ 图片显示 |
| L | 重量 | `goods.weight` 或 `skus[].weight` 或 `goodsProperty[key='产品重量']` | 商品重量，优先级：商品级>SKU级>属性 | 0 或 500g |
| M | 库存 | `skus[].quantity` | 所有SKU库存，用\|\|分隔 | 331\|\|370\|\|444 |
| N | 站点 | `goods.country` 或 `goodsProperty[key='产地']` | 站点/国家代码，优先级：country>产地>默认CN | CN 或 TH |

---

## 详细提取逻辑

### 1. 产品ID（A列）
```python
# 数据包路径
goods_id = goods.get('goodsID') or goods.get('goods_id')

# 示例数据
{
  "goods": {
    "goodsID": 872145097620  # ← 从这里提取
  }
}
```

### 2. 标题（B列）
```python
# 数据包路径（优先级从高到低）
goods_name = (
    goods.get('goodsName') or 
    goods.get('goods_name') or 
    goods.get('shareDesc') or 
    goods.get('share_desc')
)

# 示例数据
{
  "goods": {
    "goodsName": "养殖兽用奶瓶小猪奶瓶小羊人工喂养动物奶瓶专用兽用250毫升"  # ← 从这里提取
  }
}
```

### 3. 商品描述（C列）
```python
# 数据包路径
props = goods.get('goodsProperty', []) or goods.get('goods_property', [])

# 提取逻辑：遍历属性数组，构建描述
for prop in props:
    key = prop.get('key')
    values = prop.get('values', [])
    # 格式：key：value

# 示例数据
{
  "goods": {
    "goodsProperty": [
      {"key": "品牌", "values": ["其他"]},      # ← 品牌：其他
      {"key": "主材质", "values": ["其他"]},    # ← 主材质：其他
      {"key": "产地", "values": ["中国大陆"]}   # ← 产地：中国大陆
    ]
  }
}

# 生成结果
品牌：其他
主材质：其他
产地：中国大陆
```

### 4. 分类ID（D列）
```python
# 数据包路径
cat_id = goods.get('catID') or goods.get('cat_id')

# 示例数据
{
  "goods": {
    "catID": 24941  # ← 从这里提取
  }
}
```

### 5. 价格（E列）
```python
# 数据包路径
skus = goods.get('skus', []) or content.get('sku', [])

# 提取逻辑：遍历所有SKU，提取价格
for sku in skus:
    group_price = sku.get('groupPrice') or sku.get('group_price')
    # 如果是整数且>1000，除以100（分转元）
    if isinstance(group_price, int) and group_price > 1000:
        price = str(float(group_price) / 100)
    else:
        price = str(group_price)

# 示例数据
{
  "skus": [
    {"groupPrice": "21.57"},  # ← 21.57
    {"groupPrice": "15"},     # ← 15
    {"group_price": 2962}     # ← 2962 / 100 = 29.62
  ]
}

# 生成结果（用||分隔）
21.57||15||29.62
```

### 6. 规格一 & 规格一内容（F、G列）
```python
# 数据包路径
skus = goods.get('skus', [])

# 提取逻辑：
# 1. 遍历所有SKU，收集规格维度
spec_keys = {}      # {spec_key_id: spec_key_name}
spec_values = {}    # {spec_key_id: [spec_values]}

for sku in skus:
    specs = sku.get('specs', [])
    for spec in specs:
        spec_key_id = spec.get('spec_key_id')
        spec_key = spec.get('spec_key')
        spec_value = spec.get('spec_value')
        
        spec_keys[spec_key_id] = spec_key
        if spec_value not in spec_values[spec_key_id]:
            spec_values[spec_key_id].append(spec_value)

# 2. 排序规格维度ID，取第一个作为规格一
spec_key_ids = sorted(spec_keys.keys())
spec1_name = spec_keys[spec_key_ids[0]]
spec1_values = '||'.join(spec_values[spec_key_ids[0]])

# 示例数据
{
  "skus": [
    {
      "specs": [
        {
          "spec_key": "属性",                           # ← 规格一名称
          "spec_value": "250ml奶瓶1个+2个备用奶嘴",      # ← 规格一值1
          "spec_key_id": 4773
        }
      ]
    },
    {
      "specs": [
        {
          "spec_key": "属性",                           # ← 规格一名称（相同）
          "spec_value": "250毫升奶瓶【1个装+1个奶嘴】",  # ← 规格一值2
          "spec_key_id": 4773
        }
      ]
    }
  ]
}

# 生成结果
规格一: 属性
规格一内容: 250ml奶瓶1个+2个备用奶嘴||250毫升奶瓶【1个装+1个奶嘴】||...
```

### 7. 规格二 & 规格二内容（H、I列）
```python
# 提取逻辑：如果有第二个规格维度
if len(spec_key_ids) > 1:
    spec2_name = spec_keys[spec_key_ids[1]]
    spec2_values = '||'.join(spec_values[spec_key_ids[1]])
else:
    spec2_name = ''
    spec2_values = ''

# 示例数据（双规格商品，如服装）
{
  "skus": [
    {
      "specs": [
        {"spec_key": "颜色分类", "spec_value": "黑色A138", "spec_key_id": 8640},  # ← 规格一
        {"spec_key": "尺码", "spec_value": "S", "spec_key_id": 1226}              # ← 规格二
      ]
    },
    {
      "specs": [
        {"spec_key": "颜色分类", "spec_value": "黑色A138", "spec_key_id": 8640},
        {"spec_key": "尺码", "spec_value": "M", "spec_key_id": 1226}              # ← 规格二值2
      ]
    }
  ]
}

# 生成结果
规格一: 颜色分类
规格一内容: 黑色A138||浅紫色A138
规格二: 尺码
规格二内容: S||M||L||XL
```

### 9. 图片（K列）
```python
# 数据包路径（优先级从高到低）
# 方法1: viewImageData
m_urls = goods.get('viewImageData', [])

# 方法2: gallery
if not m_urls:
    gallery = goods.get('gallery', [])
    m_urls = [i.get('url') for i in gallery if isinstance(i, dict)]

# 方法3: topGallery
if not m_urls:
    top_gallery = goods.get('topGallery', [])
    m_urls = [i.get('url') for i in top_gallery if isinstance(i, dict)]

# 方法4: decoration
if not m_urls:
    decoration = goods.get('decoration', [])
    for dec in decoration[:6]:
        contents = dec.get('contents', [])
        if contents and isinstance(contents[0], dict):
            img_url = contents[0].get('img_url')
            if img_url:
                m_urls.append(img_url)

# 示例数据
{
  "goods": {
    "viewImageData": [
      "https://img.pddpic.com/mms-material-img/2021-03-05/262a9ed0-4917-48e6-96ea-c01808e38f3c.jpeg",  # ← 主图1

# K列（图片）- 嵌入的图片（不是URL文本）
程序会下载这些URL的图片，然后嵌入到Excel中显示为可视化图片
```

### 10. 重量（L列）
```python
# 数据包路径（优先级从高到低）
# 1. 商品级别重量
weight = goods.get('weight', 0)

# 2. SKU级别重量（取第一个非0的）
if not weight or weight == 0:
    for sku in skus:
        sku_weight = sku.get('weight', 0)
        if sku_weight and sku_weight != 0:
            weight = sku_weight
            break

# 3. 从goodsProperty中提取
if not weight or weight == 0:
    for prop in goods.get('goodsProperty', []):
        if '重量' in prop.get('key', ''):
            values = prop.get('values', [])
            if values and values[0] not in ['其他', '其他）', '']:
                weight = values[0]
                break

# 示例数据
{
  "goods": {
    "weight": 0,  # ← 通常为0
    "goodsProperty": [
      {"key": "产品重量", "values": ["500g"]}  # ← 从这里提取
    ]
  },
  "skus": [
    {"weight": 0}  # ← 通常也为0
  ]
}

# 生成结果
500g  # 或者 0（如果都没有有效值）
```

### 12. 站点（N列）
```python
# 数据包路径（优先级从高到低）
# 1. 从 goods.country 获取
site = goods.get('country', '')

# 2. 从 goodsProperty 中提取产地
if not site:
    for prop in goods.get('goodsProperty', []):
        key = prop.get('key', '')
        if '产地' in key or '发货地' in key:
            values = prop.get('values', [])
            if values:
                origin = values[0]
                # 映射规则
                if '中国' in origin or '大陆' in origin:
                    site = 'CN'
                elif '泰国' in origin:
                    site = 'TH'
                elif '美国' in origin:
                    site = 'US'
                elif '日本' in origin:
                    site = 'JP'
                # ... 其他国家

# 3. 默认值
if not site:
    site = 'CN'

# 示例数据
{
  "goods": {
    "country": "",  # ← 通常为空
    "goodsProperty": [
      {"key": "产地", "values": ["中国大陆"]},  # ← 从这里提取并映射为 CN
      {"key": "发货地", "values": ["河北省"]}   # ← 或从这里（省份也算CN）
    ]
  }
}

# 生成结果
CN  # 中国
# 或 TH（泰国）、US（美国）等
```

## 注意事项

1. **价格转换**：
   - 如果 `groupPrice` 是字符串（如 "21.57"），直接使用
   - 如果是整数且 > 1000（如 2962），除以100转为元（29.62）

2. **规格排序**：
   - 规格维度按 `spec_key_id` 排序
   - 第一个作为规格一，第二个作为规格二

3. **图片处理**：
   - J列和K列是**嵌入的图片**，不是文本URL
   - 每行最多嵌入6张图片（横向排列）

4. **数据包兼容性**：
   - 支持 `store.initDataObj.goods` 格式（旧格式）
   - 支持直接 `goods` + `sku` 格式（新格式）
   - 字段名支持驼峰和下划线两种命名方式
