#### 哈希算法

哈希算法是一种任意长度的输入数据映射为固定长度的输出数据的算法. 

| **算法名称** | **输出位数 (Bit)** | **Hex 字符串长度** | **特征与判断技巧**                                           |
| ------------ | ------------------ | ------------------ | ------------------------------------------------------------ |
| **MD5**      | 128 bit            | **32 位**          | 最常见。如果你看到 `32` 个字符的混淆串，盲猜 MD5 准没错。有时候会魔改为 16 位（取中间 16 个字符）。 |
| **SHA1**     | 160 bit            | **40 位**          | 比 MD5 长一点。常用于旧系统的签名。                          |
| **SHA256**   | 256 bit            | **64 位**          | 现代 Web 最常用。如果是 64 位 Hex，大概率是它。              |
| **SHA512**   | 512 bit            | **128 位**         | 超长，很少见，通常用于高安全性需求。                         |
| **HMAC**     | 512 bit            |                    | HMAC 是一种机制（Hash + 密钥）。比如 `HMAC-SHA256` 输出也是 64 位。特征是代码里一定会有个 `key` (密钥) 参与运算。 |

```python
# 密钥
import base64

from Crypto.Cipher import DES
from Crypto.Util.Padding import pad

key = b'S29jJd12jqkwl213dsq2wq321eewq1d'
# 截取前8个字节作为DES密钥
key = key[:8]

def des_encrypt(plaintext):
    # 创建DES加密对象
    cipher = DES.new(key, DES.MODE_ECB)
    # 对明文进行PKCS7填充
    padded_plaintext = pad(plaintext.encode('utf-8'), DES.block_size)
    # 加密
    cipher_text = cipher.encrypt(padded_plaintext)

    # 将加密后的数据进行Base64编码
    encode_cipher_text = base64.b64encode(cipher_text).decode('utf-8')
    # 替换 '+' 为 '%2B'
    encode_cipher_text = encode_cipher_text.replace('+', '%2B')

    return encode_cipher_text

if __name__ == '__main__':
    print(des_encrypt('hello world'))
```

```python
# 密钥
import base64

from Crypto.Cipher import AES
from Crypto.Util.Padding import pad


# 加密
def aes_encrypt(plaintext, key, iv):
    # 创建AES加密对象
    cipher = AES.new(key.encode('utf-8'), AES.MODE_CBC, iv.encode('utf-8'))
    # 对明文进行填充
    padded_plaintext = pad(plaintext.encode('utf-8'), AES.block_size)
    # 加密
    cipher_text = cipher.encrypt(padded_plaintext)
    return base64.b64encode(cipher_text)

# 解密
def aes_decrypt(ciphertext, key, iv):
    cipher_text = AES.new(key.encode('utf-8'), AES.MODE_CBC, iv.encode('utf-8'))
    plaintext = cipher_text.decrypt(base64.b64decode(ciphertext))
    return plaintext.decode('utf-8')


if __name__ == '__main__':
    key = '1234567890123123'  # 固定16位
    iv = '1234567522132132'  # 小于等于16位
    data = 'hello world'
    res = aes_encrypt(data, key, iv).decode('utf-8')
    print(res)
    res = aes_decrypt(res, key, iv)
    print(res)
```

````shell
openssl genpkey -algorithm RSA -pkeyopt rsa_keygen_bits:2048 -out rsa_private.pem
openssl rsa -pubout -in rsa_private.pem -out public_key.pem
````

```python
from Crypto.Cipher import PKCS1_OAEP
from Crypto.PublicKey import RSA

# 加密
def rsa_encrypt(pbk, plain_text):
    cipher = PKCS1_OAEP.new(RSA.import_key(pbk))
    encode_text = cipher.encrypt(plain_text.encode())
    return base64.b64encode(encode_text).decode()

# 解密
def rsa_decrypt(prk, plain_text):
    cipher = PKCS1_OAEP.new(RSA.importKey(prk))
    decode_text = cipher.decrypt(base64.b64decode(plain_text.encode()))
    return decode_text.decode()
```
